# Implementation Plan: Core Workspace Onboarding Flow

**Branch**: `020-core-workspace-onboarding-flow` | **Date**: 2026-05-14 | **Spec**: [spec.md](./spec.md)

## Summary

Add a standalone `/onboarding` route to `apps/core-platform` implementing a 3-step workspace setup flow: (1) Create Workspace (name, slug auto-gen, region, currency), (2) Choose Apps (Shops selectable, 4 others Coming Soon), (3) Review (read-only summary + "Continue to dashboard"). State lives in React; data is written to sessionStorage on completion. A returning-user guard (`core_workspace_onboarding_done`) prevents the form from showing again after completion. No new packages; 4 new icons added to the existing `Icon.tsx` system.

### 3. `lib/session.ts` — New File

```typescript
"use client";

export interface WorkspaceSetup {
  workspaceName: string;
  slug: string;
  region: string;
  currency: string;
  shopsEnabled: boolean;
}

const ONBOARDING_KEY = "core_workspace_onboarding_done";
const SETUP_KEY = "core_workspace_setup";

export function completeWorkspaceOnboarding(): void {
  if (typeof window === "undefined") return;
  sessionStorage.setItem(ONBOARDING_KEY, "true");
}

export function isWorkspaceOnboardingComplete(): boolean {
  if (typeof window === "undefined") return false;
  return sessionStorage.getItem(ONBOARDING_KEY) === "true";
}

export function saveWorkspaceSetup(data: WorkspaceSetup): void {
  if (typeof window === "undefined") return;
  sessionStorage.setItem(SETUP_KEY, JSON.stringify(data));
}
```

```typescript
function toSlug(name: string): string {
  return name.toLowerCase().trim()
    .replace(/\s+/g, "-")
    .replace(/[^a-z0-9-]/g, "")
    .replace(/-+/g, "-")
    .replace(/^-|-$/g, "");
}

// In workspace name onChange handler:
const handleNameChange = (val: string) => {
  setWorkspaceName(val);
  if (!slugManuallyEdited) setSlug(toSlug(val));
};

const handleSlugChange = (val: string) => {
  setSlugManuallyEdited(true);
  setSlug(val);
};
```
