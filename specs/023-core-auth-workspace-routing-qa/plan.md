### 2. `apps/core-platform/app/login/page.tsx`

**Before** (Server Component, button is inert):
```tsx
import Link from "next/link";
import { Button } from "@/components/ui/Button";
import { Input } from "@/components/ui/Input";

export default function LoginPage() {
  return (
    // ... form with <Button variant="primary" type="button">Sign In</Button>
  );
}
```

**After** (Client Component, button routes conditionally):
```tsx
"use client";

import Link from "next/link";
import { useRouter } from "next/navigation";
import { Button } from "@/components/ui/Button";
import { Input } from "@/components/ui/Input";
import { isWorkspaceOnboardingComplete } from "@/lib/session";

export default function LoginPage() {
  const router = useRouter();

  const handleSignIn = () => {
    router.push(isWorkspaceOnboardingComplete() ? "/workspaces" : "/onboarding");
  };

  return (
    // ... form with:
    // <Button variant="primary" type="button" onClick={handleSignIn}>
    //   Sign In
    // </Button>
  );
}
```

**Only changes**:
1. Add `"use client";` at top
2. Add `import { useRouter } from "next/navigation";`
3. Add `import { isWorkspaceOnboardingComplete } from "@/lib/session";`
4. Add `const router = useRouter();` inside component
5. Add `handleSignIn` function
6. Add `onClick={handleSignIn}` to the Button

**No changes** to form layout, Input components, Link to /register, or any visual element.
